def safe_map(series):
    return series.map(sentiment_map).astype('float64')  # NaNs remain

for c in score_cols:
    if c in df.columns:
        df[c + '_score'] = safe_map(df[c])
    else:
        # If some expected columns are missing, create an empty score column
        df[c + '_score'] = np.nan

score_score_cols = [c + '_score' for c in score_cols]

# 3) Helper: aggregated summary generator
def make_summary(group_df, name_field):
    """
    Returns:
      - avg_scores: mean per score item
      - overall: mean across all score items
      - n: number of responses
      - top_pos_comments, top_neg_comments: lists
      - keywords: common keywords
    """
    # numeric summaries
    avg_scores = group_df[score_score_cols].mean().round(3)
    overall = avg_scores.mean()
    n = len(group_df)

    # comments handling
    comments = group_df['Comments'].dropna().astype(str)
    combined_comments = ' '.join(comments.tolist())
    # simple sentiment proxy: count positive items in score columns
    pos_ratio = (group_df[score_score_cols] > 0).sum().sum() / (group_df[score_score_cols].notna().sum().sum() or 1)

    # top positive / negative by sum of scores across score columns
    group_df['sum_score'] = group_df[score_score_cols].sum(axis=1)
    top_pos = group_df.sort_values('sum_score', ascending=False).head(3)['Comments'].dropna().tolist()
    top_neg = group_df.sort_values('sum_score', ascending=True).head(3)['Comments'].dropna().tolist()

    # keywords via TF-IDF (small, fast)
    keywords = []
    if combined_comments.strip():
        vect = TfidfVectorizer(stop_words='english', ngram_range=(1,2), max_features=30)
        X = vect.fit_transform(comments)
        tfidf_sum = np.asarray(X.sum(axis=0)).ravel()
        terms = vect.get_feature_names_out()
        kw = sorted(zip(terms, tfidf_sum), key=lambda x: x[1], reverse=True)
        keywords = [t for t, s in kw[:8]]

    # drop helper column
    group_df.drop(columns=['sum_score'], inplace=True, errors='ignore')

    return {
        'name': name_field,
        'n_responses': int(n),
        'overall_score': float(np.round(float(overall),3)) if not np.isnan(overall) else None,
        'avg_scores': avg_scores.to_dict(),
        'positive_ratio': round(float(pos_ratio),3),
        'top_pos_comments': top_pos,
        'top_neg_comments': top_neg,
        'keywords': keywords
    }

# 6) Narrative generator (short paragraph)
def narrative(item):
    lines = []
    lines.append(f"Name: {item['name']}  —  Responses: {item['n_responses']}, Avg score: {item['overall_score']}")
    # top 3 strengths by avg_scores
    avg_scores = item['avg_scores']
    sorted_scores = sorted(avg_scores.items(), key=lambda kv: kv[1] if kv[1] is not None else -99, reverse=True)
    strengths = [k for k, v in sorted_scores[:3]]
    weaknesses = [k for k, v in sorted_scores[-3:]]
    lines.append("Strengths: " + ', '.join(strengths))
    lines.append("Weaknesses: " + ', '.join(weaknesses))
    if item['keywords']:
        lines.append("Common themes: " + ', '.join(item['keywords']))
    if item['top_pos_comments']:
        lines.append("Example positive comment: " + (item['top_pos_comments'][0][:200] + ('...' if len(item['top_pos_comments'][0])>200 else '')))
    if item['top_neg_comments']:
        lines.append("Example negative comment: " + (item['top_neg_comments'][0][:200] + ('...' if len(item['top_neg_comments'][0])>200 else '')))
    return '\n'.join(lines)
